# has our party size fallen below three?
if;gameStateObj.get_total_party_members() < 3
	add_minimum_number_banner
	lose_game
	t;1
elif;gameStateObj.check_dead('Sidney')
	win_game
elif;OPTIONS['Display Hints'] and gameStateObj.turncount == 1
	if;self.optionalunit.name == 'Nia'
		s;Narrator;Great!{w}{br}Now select Joel, the axe fighter.{w};auto;hint
		give_skill;Joel;Skill_Restore;no_display
		trigger_event;tutorial_nia
		tutorial_mode;Cleave
	elif;self.optionalunit.name == 'Joel'
		s;Narrator;Joel attacked both enemies at once.{w}{br}Activated skills give you powerful options in battle.{w}{br}Always be aware of what skills you have at your disposal.{w};auto;hint
		s;Narrator;Now that you've been introduced to your two new units,{w}{br}you can proceed through the rest of the chapter on your own.{w}{br}Good luck!{w};auto;hint
		tutorial_mode;0
	end
end
